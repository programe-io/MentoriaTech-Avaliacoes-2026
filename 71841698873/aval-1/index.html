// Lista de alunos let alunos = []; // Função para cadastrar aluno function cadastrarAluno(nome, nota1, nota2) { let media = calcularMedia(nota1, nota2); let status = verificarAprovacao(media); let aluno = { nome: nome, nota1: nota1, nota2: nota2, media: media, status: status }; alunos.push(aluno); } // Função para calcular média function calcularMedia(n1, n2) { return (n1 + n2) / 2; } // Função para verificar aprovação function verificarAprovacao(media) { if (media >= 7) { return "Aprovado"; } else { return "Reprovado"; } } // Função para listar alunos function listarAlunos() { for (let i = 0; i < alunos.length; i++) { console.log("Nome:", alunos[i].nome); console.log("Nota 1:", alunos[i].nota1); console.log("Nota 2:", alunos[i].nota2); console.log("Média:", alunos[i].media); console.log("Status:", alunos[i].status); console.log("----------------------"); } } // Testando o sistema cadastrarAluno("João", 8, 7); cadastrarAluno("Maria", 5, 6); cadastrarAluno("Pedro", 9, 10); // Mostrar resultados listarAlunos();
