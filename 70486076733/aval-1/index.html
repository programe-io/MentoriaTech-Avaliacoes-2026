import datetime def gerar_checklist_noticia(): print("--- Verificador de Notícias (Checklist Ético) ---") # Coleta de dados básicos titulo_noticia = input("Título da notícia analisada: ") url_noticia = input("Link da notícia: ") # Itens do Checklist (Baseado na tabela anterior) itens = [ "A fonte/URL é conhecida e confiável?", "O título é informativo (sem sensacionalismo)?", "A data de publicação é recente e clara?", "O autor é identificado e possui credibilidade?", "Existem fontes, citações ou links oficiais?", "O texto está bem escrito (sem erros grosseiros)?", "A notícia aparece em outros veículos grandes?", "As imagens/vídeos parecem autênticos e contextuais?", "Agências de checagem validaram a informação?" ] respostas = [] print("\nResponda com: OK, ALERTA ou INSUFICIENTE") for item in itens: print(f"\n> {item}") status = input("Status: ").upper() justificativa = input("Justificativa (uma linha): ") respostas.append({"item": item, "status": status, "just": justificativa}) # Lógica simples para o parecer alertas = sum(1 for r in respostas if r["status"] == "ALERTA") if alertas == 0: parecer = "CONFIÁVEL" elif alertas <= 2: parecer = "INCOMPLETA (Proceder com cautela)" else: parecer = "FALSA / ALTAMENTE SUSPEITA" # Exibição do Relatório Final print("\n" + "="*50) print(f"RELATÓRIO DE VERIFICAÇÃO - {datetime.date.today()}") print(f"NOTÍCIA: {titulo_noticia}") print("="*50) for r in respostas: print(f"[{r['status']}] {r['item']}") print(f" L Porque: {r['just']}") print("-" * 50) print(f"PARECER FINAL: {parecer}") print(f"Links consultados: {url_noticia}") print("="*50) if __name__ == "__main__": gerar_checklist_noticia()
