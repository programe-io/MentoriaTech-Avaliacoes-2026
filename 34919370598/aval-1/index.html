import random import string tsenha = 3 senha = "" for i in range(tsenha): senha += random.choice(string.digits) t = 3 while t > 0: entrada = input("Digite a senha: ") if entrada == senha: print("Acesso permitido") break else: t -= 1 print("Senha incorreta, faltam", t, "tentativas") if t == 0: print("Acesso bloqueado")
